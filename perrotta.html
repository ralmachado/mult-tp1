import matplotlib.pyplot as plt
import matplotlib.colors as clr
import numpy as np

nl = 10
nc = 10
colors = 3

randImg = np.random.rand(nl, nc, colors)
print("dim = ", randImg.shape)
plt.figure()
plt.imshow(randImg)
plt.axis('off')
plt.show()

p1 = randImg[:, :, 0]
p2 = randImg[:, :, 1]
p3 = randImg[:, :, 2]

r, c = p1.shape
while (r%16) != 0:
    p1 = np.vstack([p1, p1[-1, :]])
    p2 = np.vstack([p2, p2[-1, :]])
    p3 = np.vstack([p3, p3[-1, :]])
    r, c = p1.shape

paddedImg = np.zeros((r, c, 3))
paddedImg[:, :, 0] = p1
paddedImg[:, :, 1] = p2
paddedImg[:, :, 2] = p3
print("dim = ", paddedImg.shape)
plt.figure()
plt.imshow(paddedImg)
plt.axis('off')
plt.show()

unpaddedImg = paddedImg[:nl, :nc, :]
print("dim = ", unpaddedImg.shape)
plt.figure()
plt.imshow(unpaddedImg)
plt.axis('off')
plt.show()

Matriz de conversão RGB -> YCbCr:¶

$$\begin{bmatrix} Y \\ Cb \\ Cr \end{bmatrix} = \begin{bmatrix}0.299 & 0.587 & 0.114 \\ 0.168736 & 0.331264 & 0.5\\0.5 & 0.418688 & 0.081312\end{bmatrix}\begin{bmatrix}R \\ G \\ B\end{bmatrix} - \begin{bmatrix}0\\128\\128\end{bmatrix}$$
